Built for Developers

Use the Audionex SDK to manage agents, automate updates, integrate workflows, and control your voice AI from any application.

from audionex import Client

client = Client(api_key="your_api_key")

# Create a new agent from a prompt
agent = client.agent.create(
  prompt="You are a friendly assistant that helps users book appointments.",
  voice="alloy",
  knowledge_urls=["https://example.com/faq"]
)

# Assign a phone number
phone_number = client.phone.assign(agent_id=agent.id)
print(f"Agent is live at: {phone_number.e164}")

# List all agents
agents = client.agent.list()
print(agents)
